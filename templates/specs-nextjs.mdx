## Project Structure

```
src/
├── app/         # Next.js app router
├── server/
│   ├── api/     # tRPC routers
│   └── db/      # Prisma client
├── components/  # React components
├── hooks/       # Custom hooks
└── lib/         # Shared utilities
prisma/
└── schema.prisma  # Database schema
```

## API (tRPC)

Routers in `src/server/api/routers/`:

```typescript
// src/server/api/routers/user.ts
import { z } from 'zod';
import { router, publicProcedure, protectedProcedure } from '../trpc';

export const userRouter = router({
  getById: publicProcedure.input(z.object({ id: z.string() })).query(async ({ input, ctx }) => {
    return ctx.db.user.findUnique({ where: { id: input.id } });
  }),

  update: protectedProcedure
    .input(z.object({ name: z.string() }))
    .mutation(async ({ input, ctx }) => {
      return ctx.db.user.update({
        where: { id: ctx.user.id },
        data: { name: input.name },
      });
    }),
});
```

## Database (Prisma)

Models in `prisma/schema.prisma`:

```prisma
model User {
  id        String   @id @default(cuid())
  email     String   @unique
  name      String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  posts     Post[]
}
```

Always run `npx prisma generate` after schema changes.

## Components

Same as React guidelines. Use tRPC hooks:

```tsx
import { api } from '~/lib/api';

export function UserProfile({ userId }: { userId: string }) {
  const { data: user, isLoading } = api.user.getById.useQuery({ id: userId });

  if (isLoading) return <Spinner />;
  return <div>{user?.name}</div>;
}
```
